41 std::cout << "------------------- begin --------------------" << std::endl;

62 e += 1;

65 pi->pause();
66 err = pi->publish(b); // this publication should be silent in output
67 if (err) {
68 std::cout << "publish 2 failed " <<

75 if (err) {

81 pi->finalize();
